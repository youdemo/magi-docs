```javascript
/**
 * MultiCLI 测试工具库
 * 提供统一的日志、断言、结果追踪功能
 */

/** 颜色常量 */
const colors = {
  reset: '\x1b[0m',
  green: '\x1b[32m',
  red: '\x1b[31m',
  yellow: '\x1b[33m',
  blue: '\x1b[34m',
  cyan: '\x1b[36m',
  magenta: '\x1b[35m',
};

/**
 * 测试运行器
 * 提供统一的测试结果追踪和输出格式
 */
class TestRunner {
  constructor(name) {
    this.name = name;
    this.results = { passed: 0, failed: 0, tests: [] };
    this.startTime = Date.now();
  }

  /** 记录测试结果 */
  logTest(name, passed, details = '') {
    const symbol = passed ? '✅' : '❌';
    const color = passed ? 'green' : 'red';
    this.log(`${symbol} ${name}`, color);
    if (details) console.log(`   ${details}`);

    this.results.tests.push({ name, passed, details });
    passed ? this.results.passed++ : this.results.failed++;
  }

  /** 彩色日志 */
  log(msg, color = 'reset') {
    console.log(`${colors[color]}${msg}${colors.reset}`);
  }

  /** 章节标题 */
  logSection(title) {
    console.log('\n' + '='.repeat(80));
    this.log(`  ${title}`, 'cyan');
    console.log('='.repeat(80));
  }

  /** 完成测试并输出汇总 */
  finish() {
    const duration = ((Date.now() - this.startTime) / 1000).toFixed(2);
    const total = this.results.passed + this.results.failed;

    this.logSection('测试结果汇总');
    this.log(`⏱️  耗时: ${duration}s`, 'blue');
    this.log(`✅ 通过: ${this.results.passed}/${total}`, 'green');

    if (this.results.failed > 0) {
      this.log(`❌ 失败: ${this.results.failed}/${total}`, 'red');
      this.log('\n失败的测试:', 'red');
      this.results.tests.filter(t => !t.passed).forEach(t => {
        this.log(`  ❌ ${t.name}`, 'red');
        if (t.details) console.log(`     ${t.details}`);
      });
    }

    return this.results.failed === 0 ? 0 : 1;
  }
}

/** 导出 */
module.exports = {
  colors,
  TestRunner,
};
```

#### 1.3 添加测试运行器 (1h)

**文件**: `scripts/run-all-tests.js`

```javascript
const { spawn } = require('child_process');
const { TestRunner } = require('./test-utils');

const TEST_SUITES = {
  quick: ['test-architecture-optimization.js'],
  full: [
    'test-architecture-optimization.js',
    'e2e-architecture-test.js',
    'test-orchestrator-flow.js',
    'test-profile-e2e.js',
  ],
};

async function runTest(script) {
  return new Promise((resolve) => {
    const proc = spawn('node', [`scripts/${script}`], { stdio: 'inherit' });
    proc.on('close', (code) => resolve(code === 0));
  });
}

async function main() {
  const mode = process.argv[2] || 'quick';
  const tests = TEST_SUITES[mode] || TEST_SUITES.quick;

  const runner = new TestRunner(`MultiCLI Test Suite (${mode} mode)`);

  for (const script of tests) {
    runner.logSection(`Running: ${script}`);
    const passed = await runTest(script);
    runner.logTest(script, passed);
  }

  process.exit(runner.finish());
}

main();
```

```json
    "test:full": "node scripts/run-all-tests.js full",
    "test:quick": "node scripts/run-all-tests.js quick"
  }
}
```

#### 2.3 创建统一的 VSCode Mock

**文件**: `src/test/utils/vscode-mock.ts`

```typescript
/**
 * 统一的 VSCode API Mock
 * 所有测试共享,避免重复代码
 */
export function mockVSCodeModule() {
  const Module = require('module');
  const originalRequire = Module.prototype.require;

  Module.prototype.require = function(id: string) {
    if (id === 'vscode') {
      return {
        languages: { getDiagnostics: () => [] },
        DiagnosticSeverity: { Error: 0, Warning: 1, Information: 2, Hint: 3 },
        Uri: {
          file: (p: string) => ({ fsPath: p, path: p }),
          parse: (s: string) => ({ fsPath: s, path: s })
        },
        workspace: {
          workspaceFolders: [],
          getConfiguration: () => ({
            get: () => undefined,
            update: () => Promise.resolve()
          }),
        },
        // ... 其他 API ...
      };
    }
    return originalRequire.apply(this, arguments);
  };
}
```

**使用方式**:

```javascript
// scripts/test-example.js

mockVSCodeModule(); // 一行代码替代 30 行重复
```

### Phase 3: 中期规划 (1-2 个月)

#### 3.1 测试分层架构

```
测试金字塔:

        E2E (scripts/e2e/*.js)
       /   \
      /     \     集成测试 (scripts/integration/*.js)
     /       \
    /         \   单元测试 (src/test/unit/*.test.ts)
   /_________  \

  快速 ←--------→ 慢速
  隔离 ←--------→ 真实
```

**分层原则**:
- **单元测试** (Vitest): 快速反馈 (< 5s),测试纯逻辑
- **集成测试** (Node.js): 中速 (< 30s),测试模块协作
- **E2E 测试** (Node.js): 慢速 (< 2min),测试完整流程

#### 3.2 Mock 策略优化

**单元测试**: 完全 Mock 外部依赖
```typescript
vi.mock('../cli/adapter-factory', () => ({
  CLIAdapterFactory: class {
    sendMessage = vi.fn(() => Promise.resolve({ content: 'mocked' }))
  }
}));
```

**集成测试**: Mock CLI 进程,保留其他真实逻辑
```javascript
const fakeCLI = spawn('echo', ['{"content": "test"}']);
```

**E2E 测试**: 真实 CLI 进程
```javascript
const realCLI = spawn('claude', ['--version']);
```

#### 3.3 CI/CD 集成

## 🔧 技术决策记录

### 为什么不全面使用 Vitest?

**原因**:
1. 项目 80% 是 E2E 测试,需要真实 CLI 进程交互
2. Vitest 不适合长时间运行的集成测试 (15s+ 超时)
3. VSCode API Mock 在 Vitest 中重写成本高 (27h vs 3h)
4. 现有测试脚本稳定,重写风险大

**策略**: 混合使用
- ✅ 新的单元测试 → Vitest
- ✅ 现有 E2E 测试 → 保持 Node.js 脚本
- ✅ 逐步提取可单元化的逻辑 → 迁移到 Vitest

### 为什么保留双路径事件流?

**分析**: MessageBridge 是性能优化的副产品,不是设计缺陷

**决策**: 保持现状,长期可选择:
- 选项 A: 移到 `src/test/utils/` 明确测试专用
- 选项 B: 完全删除,测试直接用 CLIAdapterFactory
- 选项 C: 扩展为消息中间件系统

**当前推荐**: 选项 A (文档化 + 移动位置)
